# WHITE-HAT-STOAT Dockerfile
# Ubuntu-based container with all dependencies

FROM ubuntu:22.04

LABEL maintainer="WHITE-HAT-STOAT Team"
LABEL version="1.0.0"
LABEL description="Advanced Cybersecurity Command & Control Platform"

# Set environment variables
ENV DEBIAN_FRONTEND=noninteractive
ENV PYTHONUNBUFFERED=1
ENV PIP_NO_CACHE_DIR=1
ENV WORKDIR=/opt/white-hat-stoat

# Install system dependencies
RUN apt-get update && apt-get install -y \
    # Core system tools
    wget \
    curl \
    git \
    vim \
    nano \
    htop \
    net-tools \
    iputils-ping \
    traceroute \
    dnsutils \
    whois \
    nmap \
    nikto \
    # Python and build tools
    python3 \
    python3-pip \
    python3-dev \
    python3-venv \
    build-essential \
    libssl-dev \
    libffi-dev \
    # Network tools
    netcat \
    tcpdump \
    wireshark \
    scapy \
    # Database
    sqlite3 \
    postgresql-client \
    # Graphics and fonts
    libglib2.0-0 \
    libsm6 \
    libxext6 \
    libxrender-dev \
    libgomp1 \
    libxcb-xinerama0 \
    fontconfig \
    fonts-dejavu \
    # Web requirements
    wkhtmltopdf \
    # SSH and security
    openssh-client \
    openssh-server \
    sshpass \
    # Additional tools
    jq \
    xmlstarlet \
    && apt-get clean \
    && rm -rf /var/lib/apt/lists/*

# Create working directory
WORKDIR ${WORKDIR}

# Copy requirements first for better caching
COPY requirements.txt .
COPY requirements-dev.txt .
COPY requirements-extra.txt .

# Install Python dependencies
RUN pip3 install --upgrade pip setuptools wheel && \
    pip3 install -r requirements.txt && \
    pip3 install -r requirements-dev.txt || true && \
    pip3 install -r requirements-extra.txt || true

# Additional Python packages with specific dependencies
RUN pip3 install \
    cryptography \
    paramiko \
    pynput \
    selenium \
    webdriver-manager \
    telethon \
    discord.py \
    slack-sdk \
    matrix-client \
    flask-socketio \
    python-engineio \
    python-socketio \
    eventlet \
    docker \
    kubernetes \
    boto3 \
    azure-storage-blob \
    prometheus-client \
    kafka-python \
    elasticsearch \
    plotly \
    dash \
    pyqtgraph

# Copy application source
COPY . .

# Create necessary directories
RUN mkdir -p /var/log/white-hat-stoat \
    /etc/white-hat-stoat \
    /var/lib/white-hat-stoat \
    && chmod -R 755 /var/log/white-hat-stoat \
    && chmod -R 755 /var/lib/white-hat-stoat

# Set permissions
RUN chmod +x entrypoint.sh \
    && chmod +x bin/white-hat-stoat \
    || true

# Create non-root user
RUN useradd -m -s /bin/bash stoat && \
    chown -R stoat:stoat ${WORKDIR}

# Expose ports
EXPOSE 5000  # Web Dashboard
EXPOSE 5555  # Callback Server
EXPOSE 8080  # Phishing Server
EXPOSE 4444  # Keylogger Server
EXPOSE 22    # SSH
EXPOSE 443   # HTTPS

# Health check
HEALTHCHECK --interval=30s --timeout=3s --start-period=5s --retries=3 \
    CMD curl -f http://localhost:5000/health || exit 1

# Set entrypoint
ENTRYPOINT ["./entrypoint.sh"]

# Default command
CMD ["python3", "white_hat_stoat.py"]
