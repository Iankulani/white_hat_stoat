# WHITE-HAT-STOAT Makefile
# Version: 1.0.0

.PHONY: help install install-dev install-extra test lint format clean build run docker-build docker-up docker-down

# Colors
GREEN := \033[0;32m
BLUE := \033[0;34m
YELLOW := \033[1;33m
RED := \033[0;31m
NC := \033[0m

help:
	@echo "$(BLUE)🦡 WHITE-HAT-STOAT Makefile Commands$(NC)"
	@echo ""
	@echo "$(GREEN)Installation:$(NC)"
	@echo "  make install          - Install production dependencies"
	@echo "  make install-dev      - Install development dependencies"
	@echo "  make install-extra    - Install extra features"
	@echo "  make install-all      - Install all dependencies"
	@echo ""
	@echo "$(GREEN)Testing:$(NC)"
	@echo "  make test             - Run all tests"
	@echo "  make test-unit        - Run unit tests"
	@echo "  make test-integration - Run integration tests"
	@echo "  make test-coverage    - Run tests with coverage"
	@echo ""
	@echo "$(GREEN)Code Quality:$(NC)"
	@echo "  make lint             - Run linters"
	@echo "  make format           - Format code"
	@echo "  make check            - Run all checks"
	@echo ""
	@echo "$(GREEN)Docker:$(NC)"
	@echo "  make docker-build     - Build Docker image"
	@echo "  make docker-up        - Start Docker Compose"
	@echo "  make docker-down      - Stop Docker Compose"
	@echo "  make docker-logs      - View Docker logs"
	@echo ""
	@echo "$(GREEN)Build:$(NC)"
	@echo "  make build            - Build package"
	@echo "  make build-exe        - Build executable"
	@echo "  make build-docker     - Build Docker image"
	@echo ""
	@echo "$(GREEN)Cleanup:$(NC)"
	@echo "  make clean            - Clean build artifacts"
	@echo "  make clean-all        - Clean everything"

# Installation
install:
	@echo "$(BLUE)📦 Installing production dependencies...$(NC)"
	pip install -r requirements.txt

install-dev:
	@echo "$(BLUE)📦 Installing development dependencies...$(NC)"
	pip install -r requirements-dev.txt

install-extra:
	@echo "$(BLUE)📦 Installing extra dependencies...$(NC)"
	pip install -r requirements-extra.txt

install-all: install install-dev install-extra
	@echo "$(GREEN)✅ All dependencies installed$(NC)"

# Testing
test:
	@echo "$(BLUE)🧪 Running all tests...$(NC)"
	pytest tests/ -v

test-unit:
	@echo "$(BLUE)🧪 Running unit tests...$(NC)"
	pytest tests/unit/ -v

test-integration:
	@echo "$(BLUE)🧪 Running integration tests...$(NC)"
	pytest tests/integration/ -v

test-coverage:
	@echo "$(BLUE)🧪 Running tests with coverage...$(NC)"
	pytest tests/ -v --cov=white_hat_stoat --cov-report=html --cov-report=term

# Code Quality
lint:
	@echo "$(BLUE)🔍 Running linters...$(NC)"
	flake8 white_hat_stoat/ tests/
	pylint white_hat_stoat/ tests/
	mypy white_hat_stoat/ tests/
	bandit -r white_hat_stoat/

format:
	@echo "$(BLUE)🎨 Formatting code...$(NC)"
	black white_hat_stoat/ tests/
	isort white_hat_stoat/ tests/

check: lint test-coverage
	@echo "$(GREEN)✅ All checks passed$(NC)"

# Docker
docker-build:
	@echo "$(BLUE)🐳 Building Docker image...$(NC)"
	docker build -t white-hat-stoat:latest .

docker-up:
	@echo "$(BLUE)🐳 Starting Docker Compose...$(NC)"
	docker-compose up -d

docker-down:
	@echo "$(BLUE)🐳 Stopping Docker Compose...$(NC)"
	docker-compose down

docker-logs:
	@echo "$(BLUE)🐳 Viewing Docker logs...$(NC)"
	docker-compose logs -f

docker-clean:
	@echo "$(BLUE)🐳 Cleaning Docker...$(NC)"
	docker-compose down -v
	docker system prune -f

# Build
build:
	@echo "$(BLUE)📦 Building package...$(NC)"
	python -m build

build-exe:
	@echo "$(BLUE)📦 Building executable...$(NC)"
	pyinstaller --onefile --noconsole --name white-hat-stoat white_hat_stoat.py
	@echo "$(GREEN)✅ Executable built in dist/$(NC)"

build-docker:
	@echo "$(BLUE)🐳 Building Docker image...$(NC)"
	docker build -t white-hat-stoat:latest .

# Cleanup
clean:
	@echo "$(BLUE)🧹 Cleaning build artifacts...$(NC)"
	rm -rf build/ dist/ *.egg-info
	rm -rf .pytest_cache/ .coverage htmlcov/
	rm -rf __pycache__ **/__pycache__
	find . -type f -name "*.pyc" -delete

clean-all: clean
	@echo "$(BLUE)🧹 Cleaning everything...$(NC)"
	rm -rf .venv/ .env/
	rm -rf .mypy_cache/ .ruff_cache/
	rm -rf docker-images/ *.log
	docker system prune -f

# Run
run:
	@echo "$(BLUE)🚀 Running WHITE-HAT-STOAT...$(NC)"
	python white_hat_stoat.py

run-dev:
	@echo "$(BLUE)🚀 Running in development mode...$(NC)"
	export STOAT_ENVIRONMENT=development && python white_hat_stoat.py

# Quick setup
quick-setup: install docker-build
	@echo "$(BLUE)🚀 Quick setup complete!$(NC)"
	@echo "$(GREEN)Run 'make docker-up' to start$(NC)"
